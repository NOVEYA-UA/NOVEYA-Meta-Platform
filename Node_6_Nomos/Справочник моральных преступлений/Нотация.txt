#!/usr/bin/env bash
set -euo pipefail

CMD="${1:-status}"

./ops/claw.sh verify — проверка целостности/хэшей/паспортов


> Важно: push-операции должны быть publish-only: зеркала не меняют SQLite.

5) Node-режим (каждый узел отдельно)

ops/node.sh

#!/usr/bin/env bash
set -euo pipefail

ROLE="${1:-RO2}"   # RO2 | NOMOS | TELOS
BASE="${METATRON_HOME:-$HOME/metatron-8}"
DB="${DB_PATH:-$BASE/db/metatron_core.db}"

python3 "$BASE/core/local_core.py" init --role "$ROLE" --db "$DB"
python3 "$BASE/core/local_core.py" run  --role "$ROLE" --db "$DB"

Рекомендация по расписанию (cron/systemd):

RO2: run каждые 15 минут (инциденты/KPI)

NOMOS: раз в сутки (решения/контракты/претензии)

TELOS: каждые 60 минут (задачи/SLA/отчёты)
